  processTheraErrors(errorList){
    this._theraErrors=errorList;

    for(let err of this._theraErrors){
      err.tabSet=this.tabs;
      err.tabId="tab-thera";
    }


    //TODO how to update the tab titles. Doesn't seem to work in Html

    this.processErrors();
